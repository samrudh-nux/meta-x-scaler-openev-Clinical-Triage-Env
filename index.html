# ClinicalTriageEnv Enterprise — Python example import requests BASE = "https://samrudh-nux-my-healthcare-ev4u.hf.space" # 1. Start episode r = requests.post(f"{BASE}/reset", json={"task_id": "triage_hard"}).json() session_id = r["session_id"] # 2. Get multi-agent reasoning trace analysis = requests.post(f"{BASE}/analyze", json={ "session_id": session_id, "action": {"esi_level": 1, "rationale": "Acute stroke..."}, }).json() # 3. Submit to environment step = requests.post(f"{BASE}/step", json={ "session_id": session_id, "action": {"esi_level": 1, "rationale": "FAST+, LKW<2h, warfarin, ESI-1", "recommended_immediate_interventions": [ "stroke_alert", "CT_head", "INR_stat"]} }).json() # 4. Generate PDF report pdf = requests.post(f"{BASE}/report", json={"session_id": session_id}) with open("report.pdf", "wb") as f: f.write(pdf.content) # 5. AI vs AI benchmark bench = requests.post(f"{BASE}/benchmark", json={ "task_id": "triage_hard", "user_action": step["observation"] }).json() print(f"Winner: {bench['winner']}")
